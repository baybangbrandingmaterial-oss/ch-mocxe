    <!-- TÍCH HỢP HỆ THỐNG LƯU TRỮ ĐÁM MÂY (FIREBASE) -->
    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, getDocs, doc, setDoc, onSnapshot, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // Khởi tạo thông số hệ thống
        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'bay-bang-detailing';
        const apiKey = ""; // API key cho Gemini AI
        
        let app, auth, db;
        let currentUser = null;
        let allSavedDocs = [];

        // Khởi chạy Database khi trang vừa tải xong
        try {
            app = initializeApp(firebaseConfig);
            auth = getAuth(app);
            db = getFirestore(app);

            const initAuth = async () => {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            };
            initAuth();

            onAuthStateChanged(auth, (user) => {
                currentUser = user;
                if(user) {
                    // Tự động tải dữ liệu lịch sử ngầm về máy
                    fetchHistoryData();
                }
            });
        } catch(e) { console.error("Firebase init error: ", e); }

        function fetchHistoryData() {
            if(!currentUser) return;
            const historyRef = collection(db, 'artifacts', appId, 'users', currentUser.uid, 'receipts');
            onSnapshot(historyRef, (snapshot) => {
                allSavedDocs = [];
                snapshot.forEach((doc) => {
                    allSavedDocs.push({ id: doc.id, ...doc.data() });
                });
                // Sắp xếp dữ liệu mới nhất lên trên
                allSavedDocs.sort((a, b) => b.createdAt - a.createdAt);
                renderHistoryList(allSavedDocs);
            }, (error) => {
                console.error("Lỗi khi tải lịch sử: ", error);
            });
        }

        window.saveDocument = async function() {
            if (!currentUser) return alert("Hệ thống lưu trữ đang khởi động, vui lòng chờ vài giây rồi thử lại!");
            
            const btn = document.querySelector('.btn-save');
            const originalText = btn.innerText;
            btn.innerText = "⏳ Đang Lưu...";
            btn.disabled = true;

            // 1. Gom dữ liệu thông tin khách hàng
            let phieuNo = document.getElementById('phieu-no').value.trim();
            if(!phieuNo) phieuNo = "BB-" + Math.floor(1000 + Math.random() * 9000); // Tự tạo mã nếu trống
            document.getElementById('phieu-no').value = phieuNo;

            const generalData = {
                phieuNo: phieuNo,
                date: document.getElementById('current-date').value,
                customerName: document.getElementById('customer-name').value,
                carModel: document.getElementById('car-model').value,
                customerPhone: document.getElementById('customer-phone').value,
                carPlate: document.getElementById('car-plate').value.toUpperCase(),
                advisorName: document.getElementById('advisor-name').value,
                carColor: document.getElementById('car-color').value,
                carKm: document.getElementById('car-km').value,
                carFrame: document.getElementById('car-frame').value,
                carEngine: document.getElementById('car-engine').value,
                carFuel: document.getElementById('car-fuel').value,
                deliveryHour: document.getElementById('delivery-hour').value,
                deliveryMin: document.getElementById('delivery-min').value,
                deliveryDate: document.getElementById('delivery-date').value,
                createdAt: Date.now()
            };

            // 2. Gom dữ liệu bảng Kiểm Tra
            const checklist = [];
            document.querySelectorAll('.issue-row').forEach(row => {
                checklist.push({
                    checked: row.querySelector('.issue-checkbox').checked,
                    note: row.querySelector('.issue-note').value
                });
            });

            // 3. Gom dữ liệu bảng Dịch vụ
            const services = [];
            document.querySelectorAll('.service-row').forEach(row => {
                services.push({
                    name: row.querySelector('.service-name').value,
                    price: row.querySelector('.price-input').value
                });
            });

            const fullData = { ...generalData, checklist, services };

            try {
                // Lưu vào database
                const docRef = doc(db, 'artifacts', appId, 'users', currentUser.uid, 'receipts', phieuNo);
                await setDoc(docRef, fullData);
                alert("✅ Đã lưu phiếu thành công vào hệ thống!");
            } catch (error) {
                console.error(error);
                alert("❌ Lỗi khi lưu dữ liệu. Vui lòng thử lại!");
            }

            btn.innerText = originalText;
            btn.disabled = false;
        };

        window.filterHistory = function() {
            const query = document.getElementById('search-box').value.toLowerCase();
            const filtered = allSavedDocs.filter(doc => 
                (doc.carPlate && doc.carPlate.toLowerCase().includes(query)) ||
                (doc.customerPhone && doc.customerPhone.toLowerCase().includes(query)) ||
                (doc.customerName && doc.customerName.toLowerCase().includes(query)) ||
                (doc.phieuNo && doc.phieuNo.toLowerCase().includes(query))
            );
            renderHistoryList(filtered);
        };

        function renderHistoryList(docs) {
            const tbody = document.getElementById('history-list');
            tbody.innerHTML = '';
            
            if(docs.length === 0) {
                document.getElementById('empty-history').style.display = 'block';
            } else {
                document.getElementById('empty-history').style.display = 'none';
                docs.forEach(doc => {
                    const tr = document.createElement('tr');
                    tr.className = 'history-item border-b';
                    tr.onclick = () => loadSavedDocument(doc);
                    tr.innerHTML = `
                        <td class="p-2">${doc.date || 'N/A'}</td>
                        <td class="p-2 font-bold text-blue-700">${doc.carPlate || '---'}</td>
                        <td class="p-2">${doc.customerName || 'Khách vãng lai'}</td>
                        <td class="p-2">${doc.carModel || '---'}</td>
                    `;
                    tbody.appendChild(tr);
                });
            }
        }

        // TẢI DỮ LIỆU CŨ LÊN LẠI FORM
        window.loadSavedDocument = function(data) {
            // Điền Thông tin chung
            document.getElementById('phieu-no').value = data.phieuNo || '';
            document.getElementById('current-date').value = data.date || '';
            document.getElementById('customer-name').value = data.customerName || '';
            document.getElementById('car-model').value = data.carModel || '';
            document.getElementById('customer-phone').value = data.customerPhone || '';
            document.getElementById('car-plate').value = data.carPlate || '';
            document.getElementById('advisor-name').value = data.advisorName || '';
            document.getElementById('car-color').value = data.carColor || '';
            document.getElementById('car-km').value = data.carKm || '';
            document.getElementById('car-frame').value = data.carFrame || '';
            document.getElementById('car-engine').value = data.carEngine || '';
            document.getElementById('car-fuel').value = data.carFuel || '';
            document.getElementById('delivery-hour').value = data.deliveryHour || '';
            document.getElementById('delivery-min').value = data.deliveryMin || '';
            document.getElementById('delivery-date').value = data.deliveryDate || '';

            // Điền bảng Checklist
            if(data.checklist && data.checklist.length > 0) {
                const checkRows = document.querySelectorAll('.issue-row');
                data.checklist.forEach((item, index) => {
                    if(checkRows[index]) {
                        checkRows[index].querySelector('.issue-checkbox').checked = item.checked;
                        checkRows[index].querySelector('.issue-note').value = item.note || '';
                    }
                });
            }

            // Điền bảng Services
            if(data.services && data.services.length > 0) {
                const serviceRows = document.querySelectorAll('.service-row');
                data.services.forEach((item, index) => {
                    if(serviceRows[index]) {
                        serviceRows[index].querySelector('.service-name').value = item.name || '';
                        serviceRows[index].querySelector('.price-input').value = item.price || '';
                    }
                });
                window.calculateTotal(); // Cập nhật lại tổng tiền
            }

            window.closeModal('history-modal');
            alert(`Đã tải thành công hồ sơ xe ${data.carPlate || data.customerName} lên phiếu!`);
        };

        window.formatPrice = function(input) {
            let value = input.value.replace(/\D/g, '');
            input.value = value !== '' ? parseInt(value, 10).toLocaleString('vi-VN') : '';
        }

        window.calculateTotal = function() {
            let total = 0;
            document.querySelectorAll('.price-input').forEach(input => {
                let val = input.value.replace(/\D/g, '');
                if (val) total += parseInt(val, 10);
            });
            document.getElementById('total-price').value = total === 0 ? '0' : total.toLocaleString('vi-VN');
        }

        window.downloadPDF = function() {
            window.scrollTo(0, 0);
            document.querySelectorAll('.a4-page input').forEach(input => {
                if (input.type === 'checkbox') input.checked ? input.setAttribute('checked', 'checked') : input.removeAttribute('checked');
                else if (input.type === 'text') input.setAttribute('value', input.value);
            });

            const element = document.querySelector('.a4-page');
            const printBtn = document.querySelector('.btn-print');
            const originalText = printBtn.innerText;
            
            printBtn.innerText = "⏳ Đang tạo PDF...";
            printBtn.style.opacity = "0.7";
            printBtn.disabled = true;

            element.style.boxShadow = 'none';
            element.style.margin = '0';

            const opt = {
                margin: 0, 
                filename: `Phieu_Bao_Gia_${document.getElementById('car-plate').value || 'Moi'}.pdf`,
                image: { type: 'jpeg', quality: 0.98 },
                html2canvas: { scale: 2, useCORS: true, logging: false, scrollY: 0, scrollX: 0 },
                jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }
            };

            html2pdf().set(opt).from(element).save().then(() => {
                element.style.boxShadow = '';
                element.style.margin = '20px auto';
                printBtn.innerText = originalText;
                printBtn.style.opacity = "1";
                printBtn.disabled = false;
            }).catch(err => {
                element.style.boxShadow = '';
                element.style.margin = '20px auto';
                printBtn.innerText = originalText;
                printBtn.style.opacity = "1";
                printBtn.disabled = false;
                alert("Có lỗi xảy ra khi tạo PDF. Vui lòng thử lại.");
            });
        }
